67  push (std::shared_ptr<GraphType> graph,typename GraphType::VIDType u_id,typename GraphType::VIDType v_id,
68  std::array<std::array<typename GraphType::EWeightType,GraphType::NUM>,GraphType::NUM>&flow)
69  {
70  typedef typename GraphType::EWeightType EWeightType;
71 
72  if(!graph)
73  throw std::invalid_argument("push error: graph must not be nullptr!");
74  if(u_id<0||u_id>=GraphType::NUM||v_id<0||v_id>=GraphType::NUM)
75  throw std::invalid_argument("push error:id must >=0 and <N.");
76  if(!graph->vertexes.at(u_id) || !graph->vertexes.at(v_id))
77  throw std::invalid_argument("push error: vertex of id does not exist.");
78 
79  EWeightType c_f; //c_f(u,v)
80  EWeightType delt_f;//delt_f(u,v)
81 
82  if(graph->vertexes[u_id]->key==0)
83  throw std::invalid_argument("push error:u.e must >0 !");
84 
85  //*********** 获取 c_f(u,v) ***********
86  if(graph->has_edge(u_id,v_id)) //(u,v)属于E
87  c_f=graph->weight(u_id,v_id)-flow[u_id][v_id]; // c_f(u,v)=c(u,v)-f(u,v)
88  else if(graph->has_edge(v_id,u_id)) //(v,u)属于E
89  c_f=flow[v_id][u_id]; //c_f(u,v)=f(v,u)
90  else
91  throw std::invalid_argument("push error: must be (u,v) in E or (v,u) in E!");
92 
93  if(c_f==0) throw std::invalid_argument("push error:must be c_f > 0!");
94 
95  //************ 获取 /delt_f(u,v) *********
96  delt_f=std::min(graph->vertexes[u_id]->key,c_f);
97 
98  //************ 更新 flow *************
99  if(graph->has_edge(u_id,v_id)) //(u,v)属于E
100  flow[u_id][v_id]+=delt_f;
101  else if(graph->has_edge(v_id,u_id)) //(v,u)属于E
102  flow[v_id][u_id]-=delt_f;
103  else
104  throw std::invalid_argument("push error: must be (u,v) in E or (v,u) in E!");
105 
106  //************ 更新 e ***************
107  graph->vertexes[u_id]->key -= delt_f;
108  graph->vertexes[v_id]->key += delt_f;
109  }

124  template<typename GraphType> typename GraphType::VIDType min_v_at_Ef(std::shared_ptr<GraphType> graph,typename GraphType::VIDType u_id,
125  const std::array<std::array<typename GraphType::EWeightType,GraphType::NUM>,GraphType::NUM>&flow)
126  {

130 
131  if(!graph)
132  throw std::invalid_argument("min_v_at_Ef error: graph must not be nullptr!");
133  if(u_id<0||u_id>=GraphType::NUM)
134  throw std::invalid_argument("min_v_at_Ef error:id must >=0 and <N.");
135  if(!graph->vertexes.at(u_id))
136  throw std::invalid_argument("min_v_at_Ef error: vertex of id does not exist.");
137 
138  std::vector<VIDType> Ef_v; // {v:(u,v)属于Ef}
139  const MatrixType& matrix_graph=graph->matrix.matrix; //图的矩阵描述
140  auto invalid_weight=graph->matrix.invalid_weight; //无效权重
141  //************* 获取所有边(u,v)属于E_f(残留网络G_f中的边)的结点 v *************
142  for(int i=0;i<GraphType::NUM;i++)
143  {
144  auto c_u_v=matrix_graph[u_id][i];
145  if(c_u_v!=invalid_weight) //(u,v)属于E
146  if(flow[u_id][i]<c_u_v) // f(u,v)<c(u,v)
147  Ef_v.push_back(i);
148 
149  auto c_v_u=matrix_graph[i][u_id];
150  if(c_v_u!=invalid_weight) //(v,u)属于E
151  if(flow[i][u_id]>0) // f(v,u)>0
152  Ef_v.push_back(i);
153  }
154  int min_h=unlimit<int>();
155  VIDType v_id=-1;
156  for(const auto& id:Ef_v)
157  if(graph->vertexes[id]->h<min_h)
158  {
159  min_h=graph->vertexes[id]->h;
160  v_id=id;
161  }
162  return v_id;
163  }
