54  create_Gf(const std::shared_ptr<GraphType> graph,
55  std::array<std::array<typename GraphType::EWeightType,GraphType::NUM>,GraphType::NUM>&flow)
56  {
57  typedef typename GraphType::EdgeTupleType EdgeTupleType;
58  if(!graph)
59  throw std::invalid_argument("create_Gf error: graph must not be nullptr!");
60  if(graph->matrix.invalid_weight!=0)
61  throw std::invalid_argument("create_Gf error: graph invalid weight must be 0!");
62 
63  int new_invalid_weight=0;
64  std::shared_ptr<GraphType> new_graph=std::make_shared<GraphType>(new_invalid_weight);
65  //************* 创建新图的顶点 ******************
66  for(int i=0;i<GraphType::NUM;i++)
67  {
68  auto vertex=graph->vertexes.at(i);
69  if(vertex)
70  new_graph->add_vertex(vertex->key,vertex->id);
71  }
72 
73  //********** 创建新图的边 ***********************
74  std::vector<EdgeTupleType> new_edges;
75  for(int i=0;i<GraphType::NUM;i++)
76  for(int j=0;j<GraphType::NUM;j++)
77  {
78  if(i==j) continue;
79  if(!graph->vertexes[i]||!graph->vertexes[j]) continue;
80  if(graph->has_edge(i,j)) //(u,v)属于E
81  new_edges.push_back(std::make_tuple(i,j,graph->weight(i,j)-flow[i][j]));
82  else if(graph->has_edge(j,i)) //(v,u)属于E
83  new_edges.push_back(std::make_tuple(i,j,flow[j][i]));
84  }
85  new_graph->add_edges(new_edges.begin(),new_edges.end());
86  return new_graph;
87  }

177  template<typename GraphType> std::array<std::array<typename GraphType::EWeightType,GraphType::NUM>,GraphType::NUM>
178  ford_fulkerson(const std::shared_ptr<GraphType> graph,typename GraphType::VIDType src,typename GraphType::VIDType dst)
179  {
180  typedef typename GraphType::EWeightType EWeightType;
181  typedef typename GraphType::VIDType VIDType;
182  typedef std::array<std::array<EWeightType,GraphType::NUM>,GraphType::NUM> MatrixType;
183  typedef std::function<void(VIDType v_id)> ActionType;
184 
185  if(!graph)
186  throw std::invalid_argument("ford_fulkerson error: graph must not be nullptr!");
187  if(src<0||src>=GraphType::NUM||dst<0||dst>=GraphType::NUM)
188  throw std::invalid_argument("ford_fulkerson error:id must >=0 and <N.");
189  if(!graph->vertexes.at(src) || !graph->vertexes.at(dst))
190  throw std::invalid_argument("ford_fulkerson error: vertex of id does not exist.");
191 
192 
193  ActionType empty_action=[](VIDType v_id){};
194  //********** 初始化 flow 为全 0 **********
195  MatrixType flow;
196  for(int i=0;i<GraphType::NUM;i++)
197  for(int j=0;j<GraphType::NUM;j++)
198  flow[i][j]=0;
199 
200  std::shared_ptr<GraphType> Gf;
201  while(true)
202  {
203  Gf=create_Gf(graph,flow); //创建残余网络
204  //************ 寻找增广路径 *************
205  breadth_first_search<GraphType>(Gf,src,empty_action,empty_action);
206  auto path=get_path(Gf->vertexes.at(src),Gf->vertexes.at(dst));
207  if(path.size()==0) break; //不存在增广路径
208 
209  EWeightType cf=unlimit<EWeightType>();
210  for(int i=1;i<path.size();i++) //求取增广路径的残余容量
211  {
212  auto u=path[i-1];
213  auto v=path[i];
214  auto w=Gf->weight(u,v);
215  if(w<cf) cf=w;
216  }
217  //***************** 更新流矩阵 ***************
218  for(int i=1;i<path.size();i++)
219  {
220  auto u=path[i-1];
221  auto v=path[i];
222  if(graph->has_edge(u,v)) //(u,v)属于E
223  flow[u][v]+=cf;
224  else if(graph->has_edge(v,u)) //(v,u)属于E
225  flow[v][u]-=cf;
226  }
227  }
228  return flow;
229  }
230  }
