43  void bucket_sort(const Iterator begin,const Iterator end,const typename std::iterator_traits<Iterator>::value_type& min_val,
44  const typename std::iterator_traits<Iterator>::value_type& max_val)
45  {
46  assert(min_val<max_val); //确保最小值小于最大值
47  typedef typename std::iterator_traits<Iterator>::value_type T; // 迭代器指向对象的值类型
48  std::size_t real_bucket_num=10; //划分10个区间
49  std::vector<T> buckets[real_bucket_num];
50 
51  Iterator iter=begin;
52  while(iter!=end)
53  {
54  auto value=*iter;
55  auto index=(value-min_val)*real_bucket_num/(max_val-min_val); // 这里采取最简单的hash：线性分布
56  if(index<real_bucket_num)
57  buckets[index].push_back(value); //放元素到桶中
58  else
59  buckets[real_bucket_num-1].push_back(value); //最后一个区间放超大数
60  iter++;
61  }
62  std::size_t inserted_total=0;
63  for(std::size_t i=0;i<real_bucket_num;i++)
64  {
65  quick_sort<typename std::vector<T>::iterator>(buckets[i].begin(),buckets[i].end());//对某个区间排序
66  std::copy(buckets[i].begin(),buckets[i].end(),begin+inserted_total);//排序后放回原容器（因此是非原地）
67  inserted_total+=buckets[i].size();
68  }
69  }
70  }
