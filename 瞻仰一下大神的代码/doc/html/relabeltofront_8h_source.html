60  throw std::invalid_argument("discharge error: graph must not be nullptr!");
61  if(u_id<0||u_id>=GraphType::NUM)
62  throw std::invalid_argument("discharge error:id must >=0 and <N.");
63  auto vertex_u=graph->vertexes.at(u_id);
64  if(!vertex_u)
65  throw std::invalid_argument("discharge error: vertex of id does not exist.");
66  //************** 开始循环 *******************

69  auto node_v=vertex_u->N_List.current; //

73  vertex_u->N_List.current=vertex_u->N_List.head;
74  }else
75  {

77  auto vertex_v=node_v->value.lock();

79  assert(vertex_v);
80  auto v_id=vertex_v->id;
81  if(graph->has_edge(u_id,v_id))
82  c_f=graph->weight(u_id,v_id)-flow[u_id][v_id];
83  else if(graph->has_edge(v_id,u_id))
84  c_f=flow[v_id][u_id];
85  else throw std::invalid_argument("discharge error: (u,v) belongs E or (v,u) belongs E must be true!");
86 
87  //************ 根据 c_f(u,v)以及 h函数决定是否 push **************
88  if(c_f>0 &&(vertex_u->h==vertex_v->h+1))
89  push(graph,u_id,v_id,flow);
90  else
91  vertex_u->N_List.current=vertex_u->N_List.current->next;
92 
93  }

311  if(node_u!=L.head) //当u已经是L头时无需操作
312  {
313  L.front_of(node_u)->next=node_u->next;
314  node_u->next=L.head;
315  L.head=node_u;
316  }
317  }
318  node_u=node_u->next;
319  }
320  return flow;
321  }
