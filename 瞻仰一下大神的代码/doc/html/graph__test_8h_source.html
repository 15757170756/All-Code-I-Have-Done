271 TEST_F(GraphTest,test_inverse)
272 {
273  auto empty_inverse=_empty_graph->inverse();
274  auto _n_vertexes_inverse=_n_vertexes_graph->inverse();
275  auto _n_vertexes_m_edges_inverse=_n_vertexes_m_edges_graph->inverse();
276 
277  //无效边权重相同
278  EXPECT_EQ(empty_inverse->matrix.invalid_weight,_empty_graph->matrix.invalid_weight);
279  EXPECT_EQ(_n_vertexes_inverse->matrix.invalid_weight,_n_vertexes_graph->matrix.invalid_weight);
280  EXPECT_EQ(_n_vertexes_m_edges_inverse->matrix.invalid_weight,_n_vertexes_m_edges_graph->matrix.invalid_weight);
281 
282  //顶点相同
283  for(int i=0;i<G_N;i++)
284  {
285  if(empty_inverse->vertexes.at(i))
286  {
287  EXPECT_EQ(empty_inverse->vertexes.at(i)->id,_empty_graph->vertexes.at(i)->id);
288  EXPECT_EQ(empty_inverse->vertexes.at(i)->key,_empty_graph->vertexes.at(i)->key);
289  }else EXPECT_FALSE(_empty_graph->vertexes.at(i));
290 
291  if(_n_vertexes_inverse->vertexes.at(i))
292  {
293  EXPECT_EQ(_n_vertexes_inverse->vertexes.at(i)->id,_n_vertexes_graph->vertexes.at(i)->id);
294  EXPECT_EQ(_n_vertexes_inverse->vertexes.at(i)->key,_n_vertexes_graph->vertexes.at(i)->key);
295  }else EXPECT_FALSE(_n_vertexes_graph->vertexes.at(i));
296 
297  if(_n_vertexes_m_edges_inverse->vertexes.at(i))
298  {
299  EXPECT_EQ(_n_vertexes_m_edges_inverse->vertexes.at(i)->id,_n_vertexes_m_edges_graph->vertexes.at(i)->id);
300  EXPECT_EQ(_n_vertexes_m_edges_inverse->vertexes.at(i)->key,_n_vertexes_m_edges_graph->vertexes.at(i)->key);
301  }else EXPECT_FALSE(_n_vertexes_m_edges_graph->vertexes.at(i));
302  }
303  //边反向
304  EXPECT_EQ(empty_inverse->edge_tuples().size(),0);
305  EXPECT_EQ(_n_vertexes_inverse->edge_tuples().size(),0);
306  EXPECT_EQ(_n_vertexes_m_edges_inverse->edge_tuples().size(),_n_vertexes_m_edges_graph->edge_tuples().size());
307  auto edges=_n_vertexes_m_edges_graph->edge_tuples();
308  auto r_edges=_n_vertexes_m_edges_inverse->edge_tuples();
309  for(int i=0;i<edges.size();i++)
310  {
311  auto edge=edges.at(i);
312  auto r_edge=std::make_tuple(std::get<1>(edge),std::get<0>(edge),std::get<2>(edge));
313  bool founded=false;
314  for(int j=0;j<r_edges.size();j++)
315  {
316  if(r_edges.at(j)==r_edge)
317  {
318  founded=true;
319  break;
320  }
321  }
322  EXPECT_TRUE(founded);
323  }
324 }
