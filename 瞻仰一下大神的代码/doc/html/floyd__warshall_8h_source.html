114  if(!graph)

117  //**** 这里不能直接从图的矩阵描述中提取，因为这里要求 w(i,i)=0，而图的矩阵描述中，结点可能有指向自己的边

137  P[i][j]=i;
138  }
139  }
140  }

142  for(int k=0;k<GraphType::NUM;k++)
143  {
144  MatrixType newD; // D<k>
145  MatrixType newP; // P<k>
146  for(int i=0;i<GraphType::NUM;i++)
147  {
148  for(int j=0;j<GraphType::NUM;j++)
149  {
150  // D中存放的是D<k-1>,P中存放的是P<k-1>
151  EWeightType sum=0;
152  if(D[i][k]==unlimit<EWeightType>()||D[k][j]==unlimit<EWeightType>()||D[i][k]+D[k][j]>=unlimit<EWeightType>())
153  sum=unlimit<EWeightType>();
154  else
155  sum=D[i][k]+D[k][j];
156

161  }
162  else // d_i_j<k-1> > d_i_k<k-1>+d_k_j<k-1>
163  {
164  newD[i][j]= sum; //则 d_i_j<k> = d_i_k<k-1>+d_k_j<k-1>
165  newP[i][j]= P[k][j]; //则 p_i_j<k> = p_k_j<k-1>
166  }
167  }
168  }

170  P=newP;
171  }
